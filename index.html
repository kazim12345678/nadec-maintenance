function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Sheet1"); var data = JSON.parse(e.postData.contents); sheet.appendRow([ data.date, data.line, data.machine, data.type, data.duration, data.technician, data.remarks, new Date() ]); return ContentService .createTextOutput(JSON.stringify({status: "success"})) .setMimeType(ContentService.MimeType.JSON); }
